# Start redis container
docker-compose up -d redis

# Check container running
docker ps --filter name=redis

# Backend 
npm install
npx prisma generate

# Apply migrations if you want real schema changes on Neon (use with care)
npx prisma migrate deploy       # for deploying existing migrations to a remote DB

# Or for a quick push without migration history:
npx prisma db push

# Clear complete DB
npx prisma migrate reset --force --skip-seed

# Push schema to DB
npx prisma db push

# For initial filling DB
npm run seed

# Creating test users
node test-mlm.js

# Open prisma studio
npx prisma studio
